\begin{frame}{Como Funciona: Fluxo de Uso}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \vspace{0.2cm}
            \begin{tcolorbox}[m3card, title=Passo a Passo]
                \small
                \begin{enumerate}
                    \item \textbf{Login:} CPF + senha FUMP (autenticação JWT)
                    \item \textbf{Saldo:} Consulta instantânea do crédito disponível
                    \item \textbf{Recarga:} Seleciona valor e gera QR Code PIX
                    \item \textbf{Pagamento:} Escaneia QR ou copia código PIX
                    \item \textbf{Confirmação:} App polling automático até aprovação
                    \item \textbf{Crédito:} Saldo atualizado nos RUs

% --- SEGURANÇA E API ---
\begin{frame}{Integração com API FUMP e Segurança}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[m3card, title=Segurança em Primeiro Lugar]
                \small
                \begin{itemize}
                    \item \textbf{JWT Keystore:} Token em Android Keystore / EncryptedSharedPreferences
                    \item \textbf{Ownership:} API valida CPF extraído do token (nunca enviado no body)
                    \item \textbf{Rate Limits:} Tratamento de HTTP 429 com Retry-After
                    \item \textbf{HTTPS:} TLS 1.2+ obrigatório
                \end{itemize}
            \end{tcolorbox}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[m3card, title=Endpoints da API]
                \small
                \begin{tabular}{lll}
                    \toprule
                    \textbf{Método} & \textbf{Endpoint} \\
                    \midrule
                    \texttt{POST} & \texttt{/usuarios/login} \\
                    \texttt{GET}  & \texttt{/creditos/saldo} \\
                    \texttt{POST} & \texttt{/creditos/pagamento} \\
                    \texttt{GET}  & \texttt{/creditos/pagamento/:id/status} \\
                    \texttt{GET}  & \texttt{/creditos/recargas} \\
                    \texttt{GET}  & \texttt{/creditos/refeicoes} \\
                    \bottomrule
